untuk masuk ke postgre, gunakan bash:
1. docker exec -it apotekmicroservices-postgres-db-1 bash
2. psql -U useradmin -d apotek_user
3. CREATE TABLE users (
        id SERIAL PRIMARY KEY,
        name VARCHAR(100) NOT NULL,
        role VARCHAR(50) NOT NULL,
        email VARCHAR(100),
        phone VARCHAR(20),
        shift VARCHAR(20) CHECK (shift IN ('pagi', 'malam'))
    );
4. //DIBAWAH INI INPUT DATA// 
    INSERT INTO users (name, role, email, phone, shift)
    VALUES ('Ayu Rahma', 'Kasir', 'ayu@example.com', '081234567890', 'pagi');
5. SELECT * FROM users;

1. docker exec -it apotekmicroservices-mongo-db-1 mongosh -u root -p root123 --authenticationDatabase admin

3. db.createCollection("review")
